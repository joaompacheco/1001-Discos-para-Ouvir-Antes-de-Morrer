        const albumList = document.getElementById('album-list');
        const showAllBtn = document.getElementById('showAllBtn');
        const showUnheardBtn = document.getElementById('showUnheardBtn');
        const noAlbumsMessage = document.getElementById('no-albums-message');
        const storageKey = 'albumsOuvidos';

        function getHeardAlbums() {
            try {
                const heardAlbums = JSON.parse(localStorage.getItem(storageKey)) || {};
                return heardAlbums;
            } catch (e) {
                console.error("Erro ao ler do localStorage:", e);
                return {};
            }
        }

        function saveHeardAlbums(heardAlbums) {
            try {
                localStorage.setItem(storageKey, JSON.stringify(heardAlbums));
            } catch (e) {
                console.error("Erro ao salvar no localStorage:", e);
            }
        }

        function renderAlbums(showUnheardOnly) {
            albumList.innerHTML = '';
            const heardAlbums = getHeardAlbums();
            let albumsToRender = albums;

            if (showUnheardOnly) {
                albumsToRender = albums.filter(album => !heardAlbums[`${album.artist} - ${album.album}`]);
            }

            if (albumsToRender.length === 0) {
                noAlbumsMessage.classList.remove('hidden');
            } else {
                noAlbumsMessage.classList.add('hidden');
                albumsToRender.forEach(album => {
                    const albumItem = document.createElement('div');
                    albumItem.className = `album-item p-4 bg-gray-50 rounded-lg shadow-sm flex items-center justify-between space-x-2 ${showUnheardOnly ? '' : (heardAlbums[`${album.artist} - ${album.album}`] ? 'opacity-50' : '')}`;
                    albumItem.dataset.artist = album.artist;
                    albumItem.dataset.album = album.album;

                    const checkboxId = `checkbox-${album.artist}-${album.album}`.replace(/[\s\W]+/g, '-');
                    const isChecked = heardAlbums[`${album.artist} - ${album.album}`] || false;

                    albumItem.innerHTML = `
                        <div class="flex items-center space-x-3 w-3/4">
                            <input type="checkbox" id="${checkboxId}" class="form-checkbox text-purple-600 rounded-full w-5 h-5" ${isChecked ? 'checked' : ''}>
                            <label for="${checkboxId}" class="flex-1 min-w-0">
                                <h3 class="font-semibold text-gray-800 text-sm md:text-base">${album.artist}</h3>
                                <p class="text-gray-600 truncate text-xs md:text-sm">${album.album}</p>
                            </label>
                        </div>
                        <a href="https://www.youtube.com/results?search_query=${encodeURIComponent(album.artist + ' ' + album.album)}" target="_blank" class="flex-shrink-0 text-white bg-red-500 hover:bg-red-600 transition-colors duration-200 ease-in-out p-2 rounded-full shadow-md">
                            <svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" fill="currentColor" class="w-5 h-5">

                            </svg>
                        </a>
                    `;

                    const checkbox = albumItem.querySelector('input[type="checkbox"]');
                    checkbox.addEventListener('change', (e) => {
                        const newHeardAlbums = getHeardAlbums();
                        const albumKey = `${album.artist} - ${album.album}`;
                        newHeardAlbums[albumKey] = e.target.checked;
                        saveHeardAlbums(newHeardAlbums);
                        if (showUnheardOnly) {
                            albumItem.remove();
                            if (albumList.children.length === 0) {
                                noAlbumsMessage.classList.remove('hidden');
                            }
                        } else {
                            albumItem.classList.toggle('opacity-50', e.target.checked);
                        }
                    });

                    albumList.appendChild(albumItem);
                });
            }
        }
